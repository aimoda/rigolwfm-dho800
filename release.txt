A checklist for releasing a new version of rigolwfm-dho800

# sanity test the release
  make rcheck

# make sure package builds locally
  python -m build
  make clean

# update CHANGELOG.rst with recent changes
  git shortlog <last-version>..HEAD
  git commit CHANGELOG.rst -m 'update changelog'

# NOTE: Version is automatically computed as X.Y.COMMIT_COUNT by CI
# The base version (X.Y) is read from pyproject.toml
# To bump the major or minor version, update pyproject.toml:
#   git commit pyproject.toml -m 'bump version to 2.1.0'

# push changes to main branch (this triggers automatic PyPI publish)
  git push origin main

# CI will automatically:
  # 1. Compute version from pyproject.toml + commit count
  # 2. Inject version into pyproject.toml
  # 3. Build package
  # 4. Publish to PyPI

# check that published release gets sent to pypi
  open https://pypi.org/project/rigolwfm-dho800/

# check GitHub Actions workflow succeeded
  open https://github.com/aimoda/rigolwfm-dho800/actions
